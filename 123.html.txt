    <script>
        /**
         * Poster Rating Logic
         * Checked and validated for 38 teams.
         */
        const listContainer = document.getElementById('list');
        const dataStore = {};
        const TOTAL_TEAMS = 38;

        // 1. Setup the UI on Page Load
        function initializePanel() {
            for (let i = 1; i <= TOTAL_TEAMS; i++) {
                // Prepare score object for each team
                dataStore[i] = { c1: 0, c2: 0, c3: 0 };
                
                const card = document.createElement('div');
                card.className = 'team-card';
                card.innerHTML = `
                    <div class="team-title">TEAM ${i}</div>
                    
                    <span class="label l-1">Novelty</span>
                    <div class="grid">${createRatingButtons(i, 1)}</div>
                    
                    <span class="label l-2">Poster Design</span>
                    <div class="grid">${createRatingButtons(i, 2)}</div>
                    
                    <span class="label l-3">Presentation</span>
                    <div class="grid">${createRatingButtons(i, 3)}</div>
                `;
                listContainer.appendChild(card);
            }
        }

        // 2. Button Generation Helper
        function createRatingButtons(teamId, catId) {
            let btns = '';
            for (let v = 1; v <= 10; v++) {
                btns += `<button class="btn b-${teamId}-${catId}" onclick="updateScore(${teamId}, ${catId}, ${v}, this)">${v}</button>`;
            }
            return btns;
        }

        // 3. Score Tracking Logic
        function updateScore(teamId, catId, value, element) {
            // Update dataStore
            dataStore[teamId]['c' + catId] = value;

            // Update UI Styling for the specific team and category
            const categoryButtons = document.querySelectorAll(`.b-${teamId}-${catId}`);
            categoryButtons.forEach(btn => btn.classList.remove('active-1', 'active-2', 'active-3'));
            
            element.classList.add('active-' + catId);

            // Optional: Provide haptic vibration on mobile
            if (navigator.vibrate) navigator.vibrate(12);
        }

        // 4. Excel Export Functionality
        function generateExcel() {
            const judgeName = document.getElementById('jName').value.trim();
            if (!judgeName) {
                alert("Please enter the Judge's Name before downloading.");
                return;
            }

            const rows = [["TEAM NUMBER", "NOVELTY", "POSTER", "PRESENTATION", "TOTAL"]];
            
            for (let i = 1; i <= TOTAL_TEAMS; i++) {
                const s = dataStore[i];
                // Check if all fields for a team are filled (score > 0)
                if (s.c1 === 0 || s.c2 === 0 || s.c3 === 0) {
                    alert(`Team ${i} is incomplete. Please ensure all teams have scores.`);
                    return;
                }
                const total = s.c1 + s.c2 + s.c3;
                rows.push([i, s.c1, s.c2, s.c3, total]);
            }

            try {
                // Formatting the Excel sheet data
                const finalData = [
                    ["JUDGING REPORT"],
                    ["JUDGE NAME:", judgeName.toUpperCase()],
                    ["DATE:", new Date().toLocaleDateString()],
                    [],
                    ...rows
                ];

                const ws = XLSX.utils.aoa_to_sheet(finalData);
                const wb = XLSX.utils.book_new();
                XLSX.utils.book_append_sheet(wb, ws, "Poster Ratings");

                // Download the file
                XLSX.writeFile(wb, `Judging_${judgeName.replace(/\s+/g, '_')}.xlsx`);
            } catch (err) {
                alert("Error generating Excel file: " + err.message);
            }
        }

        // Run Initialization
        initializePanel();
    </script>
